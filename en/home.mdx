<script
  dangerouslySetInnerHTML={{
    __html: `
      (function() {
        'use strict';
        
        function centerCardTitles() {
          const cardGroups = document.querySelectorAll('.card-group-centered');
          cardGroups.forEach(function(group) {
            const titles = group.querySelectorAll('[data-component-part="card-title"]');
            titles.forEach(function(title) {
              title.style.textAlign = 'center';
            });
          });
        }
        
        function updateTagColors() {
          const pills = document.querySelectorAll('.nav-tag-pill');
          pills.forEach(function(pill) {
            const text = pill.textContent.trim().toUpperCase();
            if (text === 'NEW') {
              pill.classList.add('tag-new');
            } else if (text === 'BETA') {
              pill.classList.remove('tag-new');
            }
          });
        }

        function init() {
          updateTagColors();
          centerCardTitles();
        }

        // Run on initial load
        if (document.readyState === 'loading') {
          document.addEventListener('DOMContentLoaded', init);
        } else {
          init();
        }
        
        // Use requestAnimationFrame for efficient one-time delayed execution
        requestAnimationFrame(function() {
          setTimeout(init, 500);
        });

        // Watch for DOM changes in main content area only
        var mainContent = document.querySelector('.main-container-mobile');
        if (mainContent) {
          var observer = new MutationObserver(function(mutations) {
            var shouldUpdate = false;
            mutations.forEach(function(mutation) {
              if (mutation.addedNodes.length > 0) {
                mutation.addedNodes.forEach(function(node) {
                  if (node.nodeType === 1 && 
                      (node.classList && (node.classList.contains('nav-tag-pill') || node.classList.contains('card-group-centered')) ||
                       node.querySelector && (node.querySelector('.nav-tag-pill') || node.querySelector('.card-group-centered')))) {
                    shouldUpdate = true;
                  }
                });
              }
            });
            if (shouldUpdate) {
              requestAnimationFrame(init);
            }
          });
          
          observer.observe(mainContent, { 
            childList: true, 
            subtree: true
          });
        }
      })();
    `,
  }}
