### Usage

<CodeGroup>
```bash curl
curl -X POST 'https://portal.sqd.dev/datasets/0g-testnet/stream' \
  -H 'Content-Type: application/json' \
  -d '{
    "type": "evm",
    "fromBlock": 10000000,
    "toBlock": 10001000,
    "fields": {
      "block": { "number": true, "hash": true },
      "log": { "address": true, "topics": true, "data": true }
    },
    "logs": [{
      "address": ["0x0000000000000000000000000000000000000000"]
    }]
  }'
```

```typescript TypeScript
import { createEvmPortalSource, EvmQueryBuilder } from "@sqd-pipes/pipes/evm";

const queryBuilder = new EvmQueryBuilder()
  .addFields({
    block: { number: true, hash: true },
    log: { address: true, topics: true, data: true },
  })
  .addLog({
    request: { address: ["0x0000000000000000000000000000000000000000"] },
    range: { from: 10000000 },
  });

const source = createEvmPortalSource({
  portal: "https://portal.sqd.dev/datasets/0g-testnet",
  query: queryBuilder,
});
```

```python Python
import requests

url = "https://portal.sqd.dev/datasets/0g-testnet/stream"
headers = {"Content-Type": "application/json"}
payload = {
    "type": "evm",
    "fromBlock": 10000000,
    "toBlock": 10001000,
    "fields": {
        "block": {"number": True, "hash": True},
        "log": {"address": True, "topics": True, "data": True}
    },
    "logs": [{
        "address": ["0x0000000000000000000000000000000000000000"]
    }]
}

response = requests.post(url, headers=headers, json=payload)
data = response.json()
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "net/http"
)

func main() {
    url := "https://portal.sqd.dev/datasets/0g-testnet/stream"
    payload := map[string]interface{}{
        "type":      "evm",
        "fromBlock": 10000000,
        "toBlock":   10001000,
        "fields": map[string]interface{}{
            "block": map[string]bool{"number": true, "hash": true},
            "log":   map[string]bool{"address": true, "topics": true, "data": true},
        },
        "logs": []map[string]interface{}{
            {"address": []string{"0x0000000000000000000000000000000000000000"}},
        },
    }

    jsonData, _ := json.Marshal(payload)
    resp, _ := http.Post(url, "application/json", bytes.NewBuffer(jsonData))
    defer resp.Body.Close()
}
```
