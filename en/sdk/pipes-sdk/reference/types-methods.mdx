## Types

### DecodedEvent

```ts
interface DecodedEvent<T, F = unknown> {
  blockNumber: number;
  timestamp: Date;
  contract: string;
  transactionHash: string;
  logIndex: number;
  rawEvent: {
    address: string;
    data: string;
    topics: string[];
  };
  event: T;
  factory?: {
    event: F;
  };
}
```

### EvmPortalData

```ts
interface EvmPortalData<Q> {
```

### Context

```ts
interface Context {
  logger: Logger;
  profiler: Profiler;
}
```

### Logger

```ts
interface Logger {
  info(message: string | object, ...args: any[]): void;
  warn(message: string | object, ...args: any[]): void;
  error(message: string | object, ...args: any[]): void;
}
```

### Profiler

```ts
interface Profiler {
  start(id: string): Span;
}

interface Span {
  end(): void;
}
```

### BlockCursor

```ts
interface BlockCursor {
  number: number;
  hash: string;
}
```

## Pipe Methods

### pipe

Chain a transformer.

```ts
source.pipe(transformer);
```

### pipeComposite

Chain multiple transformers.

```ts
source.pipeComposite({
  transfers: transformer1,
  swaps: transformer2,
});
```

### pipeTo

Connect to a target.

```ts
source.pipeTo(target);
```

<Note>
`pipeTo` is a terminal step. You cannot continue piping after calling this method.
</Note>
